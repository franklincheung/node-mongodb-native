1 var BaseCommand = require('./base_command').BaseCommand,
  2   BinaryParser = require('../bson/binary_parser').BinaryParser,
  3   inherits = require('sys').inherits;
  4 
  5 /**
  6   Get More Document Command
  7 @constructor
  8 @augments BaseCommand
  9 @param db Object representing the database
 10 @param collectionName String representing the name of the collection
 11 @param numberToReturn
 12 @param cursorId
 13 */
 14 var GetMoreCommand = exports.GetMoreCommand = function(db, collectionName, numberToReturn, cursorId) {
 15   BaseCommand.call(this);
 16 
 17   this.collectionName = collectionName;
 18   this.numberToReturn = numberToReturn;
 19   this.cursorId = cursorId;
 20   this.db = db;
 21 };
 22 
 23 inherits(GetMoreCommand, BaseCommand);
 24 
 25 /**
 26 Returns the opcode
 27 */
 28 GetMoreCommand.prototype.getOpCode = function() {
 29   return BaseCommand.OP_GET_MORE;
 30 };
 31 
 32 /**
 33 Generates and returns the command string
 34 */
 35 GetMoreCommand.prototype.getCommand = function() {
 36   // Generate the command string
 37   return BinaryParser.fromInt(0) + BinaryParser.encode_cstring(this.collectionName) + BinaryParser.fromInt(this.numberToReturn) + this.db.bson_serializer.BSON.encodeLong(this.cursorId);
 38 };
