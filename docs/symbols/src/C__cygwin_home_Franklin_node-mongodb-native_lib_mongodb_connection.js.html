1 var net = require('net'),
  2   EventEmitter = require("events").EventEmitter,
  3   BinaryParser = require('./bson/binary_parser').BinaryParser,
  4   inherits = require('sys').inherits;
  5 /**@constructor
  6 Object representing a server connection
  7 @param host
  8 @param port
  9 @param autoReconnect
 10 */
 11 var Connection = exports.Connection = function(host, port, autoReconnect) {
 12   this.host = host;
 13   this.port = port;
 14   this.autoReconnect = autoReconnect;
 15   this.drained = true;
 16   // Reconnect buffer for messages
 17   this.messages = [];
 18   // Message sender
 19   var self = this;
 20   // Status messages
 21   this.sizeOfMessage = 0;
 22   this.bytesRead = 0;
 23   this.buffer = '';
 24   this.stubBuffer = '';
 25 };
 26 
 27 inherits(Connection, EventEmitter);
 28 
 29 // Functions to open the connection
 30 Connection.prototype.open = function() {
 31   // Assign variable to point to local scope object
 32   var self = this;
 33   // Create the associated connection
 34   this.connection = net.createConnection(this.port, this.host);
 35   // Set up the net client
 36   this.connection.setEncoding("binary");
 37   // Add connnect listener
 38   this.connection.addListener("connect", function() {
 39     this.setEncoding("binary");
 40     this.setTimeout(0);
 41     this.setNoDelay();
 42     self.emit("connect");
 43   });
 44   
 45   this.connection.addListener("error", function(err) {
 46     self.emit("error", err);
 47   });
 48   
 49   // Add a close listener
 50   this.connection.addListener("close", function() {
 51     self.emit("close");
 52   });
 53   
 54   // Listener for receive data
 55   this.receiveListener = function(result) {
 56     // Check if we have an unfinished message
 57     if(self.bytesRead > 0 && self.sizeOfMessage > 0) {
 58       // Calculate remaing bytes to fetch
 59       var remainingBytes = self.sizeOfMessage - self.bytesRead;
 60       // Check if we have multiple packet messages and save the pieces otherwise emit the message
 61       if(remainingBytes > result.length) {
 62         self.buffer = self.buffer + result; self.bytesRead = self.bytesRead + result.length;
 63       } else {
 64         // Cut off the remaining message
 65         self.buffer = self.buffer + result.substr(0, remainingBytes);
 66         // Emit the message
 67         self.emit("data", self.buffer);
 68         // Reset the variables
 69         self.buffer = ''; self.bytesRead = 0; self.sizeOfMessage = 0;
 70         // If message is longer than the current one, keep parsing
 71         if(remainingBytes < result.length) {
 72           self.receiveListener(result.substr(remainingBytes, (result.length - remainingBytes)));
 73         }
 74       }
 75     } else {
 76       if(self.stubBuffer.length > 0) {
 77         result = self.stubBuffer + result;
 78         self.stubBuffer = '';
 79       }
 80 
 81       if(result.length > 4) {
 82         var sizeOfMessage = BinaryParser.toInt(result.substr(0, 4));
 83         // We got a partial message, store the result and wait for more
 84         if(sizeOfMessage > result.length) {
 85           self.buffer = self.buffer + result; self.bytesRead = result.length; self.sizeOfMessage = sizeOfMessage;
 86         } else if(sizeOfMessage == result.length) {
 87           self.emit("data", result);
 88         } else if(sizeOfMessage < result.length) {
 89           self.emit("data", result.substr(0, sizeOfMessage));
 90           self.receiveListener(result.substr(sizeOfMessage, (result.length - sizeOfMessage)));
 91         }
 92       } else {
 93         self.stubBuffer = result;
 94       }
 95     }
 96   };
 97 
 98   // Add a receieved data connection
 99   this.connection.addListener("data", this.receiveListener);
100 };
101 
102 Connection.prototype.close = function() {
103   if(this.connection) this.connection.end();
104 };

106 Connection.prototype.send = function(command) {
107   var self = this;
108   // Check if the connection is closed
109   try {
110     this.connection.write(command.toBinary(), "binary");
111   } catch(err) {
112     // Check if the connection is closed
113     if(this.connection.readyState != "open" && this.autoReconnect) {
114       // Add the message to the queue of messages to send
115       this.messages.push(command);
116       // Initiate reconnect if no current running
117       if(this.connection.currently_reconnecting == null) {
118         this.connection.currently_reconnecting = true;
119         // Create the associated connection
120         var new_connection = net.createConnection(this.port, this.host);
121         // Set up the net client
122         new_connection.setEncoding("binary");
123         // Add connnect listener
124         new_connection.addListener("connect", function() {
125           this.setEncoding("binary");
126           this.setTimeout(0);
127           this.setNoDelay();
128           // Add the listener
129           this.addListener("data", self.receiveListener);
130           // assign the new ready connection
131           self.connection = this;
132           // send all the messages
133           while(self.messages.length > 0) {
134             this.write(self.messages.shift().toBinary(), "binary");
135           }
136         });
137       }
138    } else {
139       throw err;
140     }
141   }
142 };

166 // Server pair object used to support a failover connection set
167 var ServerPair = exports.ServerPair = function(leftServer, rightServer) {
168   if(leftServer == null || rightServer == null || !(leftServer instanceof Server) || !(rightServer instanceof Server)) {
169     throw Error("Both left/right must be defined and off the type Server");
170   }
171   this.leftServer = leftServer;
172   this.rightServer = rightServer;
173   // Containst the master server entry
174   this.master = null;
175   this.target = null;
176   // Setters and getters
177   this.__defineGetter__("autoReconnect", function() {
178     if(this.target != null) return this.target.autoReconnect;
179     if(this.masterConnection != null) return this.masterConnection.autoReconnect;
180   });
181   this.__defineGetter__("masterConnection", function() {
182     if(this.target != null && this.target instanceof Server) return this.target.masterConnection;
183     if(this.leftServer.master) return this.leftServer.masterConnection;
184     if(this.rightServer.master) return this.rightServer.masterConnection;
185     return null;
186   });
187 };

197 // Server cluster (one master and multiple read slaves)
198 var ServerCluster = exports.ServerCluster = function(servers) {
199   // Containst the master server entry
200   this.master = null;
201   this.target = null;
202 
203   if(servers.constructor != Array || servers.length == 0) {
204     throw Error("The parameter must be an array of servers and contain at least one server");
205   } else if(servers.constructor == Array || servers.length > 0) {
206     var count = 0;
207     servers.forEach(function(server) {
208       if(server instanceof Server) count = count + 1;
209     });
210 
211     if(count < servers.length) {
212       throw Error("All server entries must be of type Server");
213     } else {
214       this.servers = servers;
215     }
216   }
217   // Setters and getters
218   this.__defineGetter__("autoReconnect", function() {
219     if(this.target != null) return this.target.autoReconnect;
220     if(this.masterConnection != null) return this.masterConnection.autoReconnect;
221   });
222   this.__defineGetter__("masterConnection", function() {
223     // Allow overriding to a specific connection
224     if(this.target != null && this.target instanceof Server) {
225       return this.target.masterConnection;
226     } else {
227       var finalServer = null;
228       this.servers.forEach(function(server) {
229         if(server.master == true) finalServer = server;
230       });
231       return finalServer != null ? finalServer.masterConnection : finalServer;
232     }
233   });
234 };

237   this.target = target;
