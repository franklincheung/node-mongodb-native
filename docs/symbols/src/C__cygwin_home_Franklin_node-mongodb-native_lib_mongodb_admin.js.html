1 var Collection = require('./collection').Collection,
  2   Cursor = require('./cursor').Cursor,

 10 Admin.prototype.profilingLevel = function(callback) {
 11   var command = {profile:-1};
 12   this.db.executeDbCommand(command, function(err, doc) {
 13     doc = doc.documents[0];
 14     if(err == null && (doc.ok == 1 || doc.was.constructor == Numeric)) {
 15       var was = doc.was;
 16       if(was == 0) {
 17         callback(null, "off");
 18       } else if(was == 1) {
 19         callback(null, "slow_only");
 20       } else if(was == 2) {
 21         callback(null, "all");
 22       } else {
 23         callback(new Error("Error: illegal profiling level value " + was), null);
 24       }
 25     } else {
 26       err != null ? callback(err, null) : callback(new Error("Error with profile command"), null);
 27     }
 28   });
 29 };

 31 Admin.prototype.setProfilingLevel = function(level, callback) {
 32   var command = {};
 33   var profile = 0;
 34   if(level == "off") {
 35     profile = 0;
 36   } else if(level == "slow_only") {
 37     profile = 1;
 38   } else if(level == "all") {
 39     profile = 2;
 40   } else {
 41     callback(new Error("Error: illegal profiling level value " + level));
 42     return;
 43   }
 44   command['profile'] = profile;
 45 
 46   this.db.executeDbCommand(command, function(err, doc) {
 47     doc = doc.documents[0];
 48     if(err == null && (doc.ok == 1 || doc.was.constructor == Numeric)) {
 49       callback(null, level);
 50     } else {
 51       err != null ? callback(err, null) : callback(new Error("Error with profile command"), null);
 52     }    
 53   });
 54 };

 56 Admin.prototype.profilingInfo = function(callback) {
 57   new Cursor(this.db, new Collection(this.db, DbCommand.SYSTEM_PROFILE_COLLECTION), {}).toArray(function(err, items) {
 58     callback(err, items);
 59   });  
 60 };

 62 Admin.prototype.validatCollection = function(collectionName, callback) {
 63   var command = {validate: collectionName};
 64   this.db.executeDbCommand(command, function(err, doc) {
 65     doc = doc.documents[0];
 66 
 67     if(err != null) {
 68       callback(err, null);
 69     } else if(doc.ok == 0) {
 70       callback(new Error("Error with validate command"), null);
 71     } else if(doc.result.constructor != String) {
 72       callback(new Error("Error with validation data"), null);
 73     } else if(doc.result.match(/exception|corrupt/) != null) {
 74       callback(new Error("Error: invalid collection " + collectionName), null);
 75     } else {
 76       callback(null, doc);
 77     }
 78   });
 79 };
