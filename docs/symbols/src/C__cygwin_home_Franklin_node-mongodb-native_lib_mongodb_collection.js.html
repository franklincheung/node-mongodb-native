 13 var formatSortValue = function(sortDirection) {
 14   var value = ("" + sortDirection).toLowerCase();
 15   if(value == 'ascending' || value == 'asc' || value == 1) return 1;
 16   if(value == 'descending' || value == 'desc' || value == -1 ) return -1;
 17   throw new Error("Illegal sort clause, must be of the form " +
 18     "[['field1', '(ascending|descending)'], ['field2', '(ascending|descending)']]");
 19 };

 21 var formattedOrderClause = function(sortValue) {
 22   var orderBy = {};
 23   var self = this;
 24 
 25   if(sortValue instanceof Array) {
 26     sortValue.forEach(function(sortElement) {
 27       if(sortElement.constructor == String) {
 28         orderBy[sortElement] = 1;
 29       } else {
 30         orderBy[sortElement[0]] = formatSortValue(sortElement[1]);
 31       }
 32     });
 33   } else if(sortValue.constructor == String) {
 34     orderBy[sortValue] = 1;
 35   } else {
 36     throw new Error("Illegal sort clause, must be of the form " +
 37       "[['field1', '(ascending|descending)'], ['field2', '(ascending|descending)']]");
 38   }
 39   return orderBy;
 40 };

 66 Collection.prototype.checkCollectionName = function(collectionName) {
 67   if (typeof collectionName != 'string')
 68     throw Error("collection name must be a String");
 69 
 70   if (!collectionName || collectionName.indexOf('..') != -1)
 71     throw Error("collection names cannot be empty");
 72 
 73   if (collectionName.indexOf('$') != -1 && collectionName.match(/((^\$cmd)|(oplog\.\$main))/) == null)
 74     throw Error("collection names must not contain '$'");
 75 
 76   if (collectionName.match(/^\.|\.$/) != null)
 77     throw Error("collection names must not start or end with '.'");
 78 };

123 Collection.prototype.insertAll = function(docs, options, callback) {
124   var error= null;
125   var args = Array.prototype.slice.call(arguments, 1);
126   callback = args.pop();
127   options = args.length ? args.shift() : {};
128   
129   try {
130     // List of all id's inserted
131     var objects = [];
132     // Create an insert command
133     var insertCommand = new InsertCommand(this.db, this.db.databaseName + "." + this.collectionName);
134     // Add id to each document if it's not already defined
135     for(var index = 0; index < docs.length; index++) {
136       var doc = docs[index];
137       // Add the id to the document
138       var id = doc["_id"] == null ? this.pkFactory.createPk() : doc["_id"];
139       doc['_id'] = id;
140       // Insert the document      
141       insertCommand.add(doc);
142       objects.push(doc);
143     }
144     // Execute the command
145     this.db.executeCommand(insertCommand);
146     // If safe is defined check for error message
147     if(options.safe) {
148       this.db.error(function(err, doc) {
149         if(doc[0].ok == false) {
150           if(callback != null) callback(new Error(doc[0].err), null);
151         } else {
152           if(callback != null) callback(null, objects);
153         }
154       });              
155     } else {
156       if(callback != null) callback(null, objects);
157     }
158   } catch(err) {
159     error= err;
160     // Return error message if needed
161     if(callback != null)  {
162       if( error ) callback(error, null);
163       else callback(null, objects);
164     }      
165   }
166 };

181 /**
182   Update a single document in this collection.
183     spec - a associcated array containing the fields that need to be present in
184       the document for the update to succeed
185 
186     document - an associated array with the fields to be updated or in the case of
187       a upsert operation the fields to be inserted.
188 
189   Options:
190     upsert - true/false (perform upsert operation)
191     multi - true/false (update all documents matching spec)
192     safe - true/false (perform check if the operation failed, required extra call to db)
193 **/
194 Collection.prototype.update = function(spec, document, options, callback) {
195   var args = Array.prototype.slice.call(arguments, 2);
196   callback = args.pop();
197   options = args.length ? args.shift() : null;
198 
199   try {
200     var safe = options == null || options.safe == null || options.safe == false ? false : true;
201     // Create update command
202     var updateCommand = new UpdateCommand(this.db, this.db.databaseName + "." + this.collectionName, spec, document, options);
203     // Execute command
204     this.db.executeCommand(updateCommand);
205     // If safe, we need to check for successful execution
206     if(safe) {
207       this.db.error(function(err, doc) {
208         if(doc[0].updatedExisting == false) {
209           if(callback != null) callback(new Error("Failed to update document"), null);
210         } else {
211           if(callback != null) callback(null, document);
212         }
213       });
214     } else {
215       // Call back with ok if no error found
216       if(callback != null) callback(null, document);
217     }
218   } catch(err) {
219     if(callback != null) callback(err, null);
220   }

260 /**
261   Fetch and update a collection
262   query:        a filter for the query
263   sort:         if multiple docs match, choose the first one in the specified sort order as the object to manipulate
264   update:       an object describing the modifications to the documents selected by the query
265   options:
266     remove:   set to a true to remove the object before returning
267     new:      set to true if you want to return the modified object rather than the original. Ignored for remove.
268     upsert:       true/false (perform upsert operation)
269 **/
270 Collection.prototype.findAndModify = function(query, sort, update, options, callback) {
271   var args = Array.prototype.slice.call(arguments, 1);
272   callback = args.pop();
273   sort = args.length ? args.shift() : [];
274   update = args.length ? args.shift() : null;
275   options = args.length ? args.shift() : {};
276 
277   // Format the sort object
278   var sort_object = formattedOrderClause(sort);
279   // Unpack the options
280   var queryOptions = QueryCommand.OPTS_NONE;
281   // Build the query object for the findAndModify
282   var queryObject = {
283     'findandmodify': this.collectionName,
284     'query':query,
285     'sort':sort_object
286   }
287 
288   queryObject['new'] = options['new'] ? 1 : 0;
289   queryObject['remove'] = options.remove ? 1 : 0;
290   queryObject['upsert'] = options.upsert ? 1 : 0;
291 
292   // Set up the update if it exists
293   if(update) queryObject['update'] = update;
294 
295   // Set up the sort
296   if(!Array.isArray(sort) && sort.length == 0) queryObject['sort'] = sort_object;
297 
298   if(options.remove) delete queryObject['update'];
299 
300   // Execute command
301   this.db.executeDbCommand(queryObject, function(err, doc) {
302     err ? callback(err, doc) : callback(err, doc.documents[0].value);
303   });
304 }

306 /*
307 various argument possibilities
308   1 callback
309   2 selector, callback,
310   2 callback, options  // really?!
311   3 selector, fields, callback
312   3 selector, options, callback
313   4,selector, fields, options, callback
314   5 selector, fields, skip, limit, callback
315   6 selector, fields, skip, limit, timeout, callback       
316 */
317 Collection.prototype.find = function() {
318   var options,
319       len = arguments.length,
320       selector = (len > 1) ? arguments[0] : {},
321       fields = (len > 2) ? arguments[1] : undefined,
322       callback = arguments[len-1];
323   
324   if(len == 2 && typeof arguments[0] == 'function') {
325     selector = {}; options = arguments[1]; callback = arguments[0];
326   }
327   
328   if(len == 3){ // backwards compat for options object
329     var test = ['limit','sort','fields','skip','hint','explain','snapshot','timeout'],
330         idx = 0, l = test.length, is_option = false;    
331     while(!is_option && idx < l) if(test[idx] in fields ) is_option = true; else idx++;
332     options = is_option ? fields : {};
333     if(is_option) fields = undefined;
334   }
335   if(len == 4) options = arguments[2];
336   
337   if(options && options.fields){
338     fields = {};
339     if(options.fields.constructor == Array){
340       if(options.fields.length == 0) fields['_id'] = 1;
341       else for(i = 0, l = options.fields.length; i < l; i++) fields[options.fields[i]] = 1;
342     }
343     else fields = options.fields;
344   }
345   if(!options) options = {};
346   
347   options.skip = len > 4 ? arguments[2] : options.skip ? options.skip : 0;
348   options.limit = len > 4 ? arguments[3] : options.limit ? options.limit : 0;
349   options.hint = options.hint != null ? this.normalizeHintField(options.hint) : this.internalHint;
350   options.timeout = len == 6 ? arguments[4] : options.timeout ? options.timeout : undefined;
351     
352   var o = options;
353   callback(null, new Cursor(this.db, this, selector, fields, o.skip, o.limit, o.sort, o.hint, o.explain, o.snapshot, o.timeout)); 
354 };

356 Collection.prototype.normalizeHintField = function(hint) {
357   var finalHint = null;
358   // Normalize the hint parameter
359   if(hint != null && hint.constructor == String) {    
360     finalHint = {};
361     finalHint[hint] = 1;
362   } else if(hint != null && hint.constructor == Object) {
363     finalHint = {};
364     for(var name in hint) { finalHint[name] = hint[name]; }
365   } else if(hint != null && hint.constructor == Array) {
366     finalHint = {};
367     hint.forEach(function(param) { finalHint[param] = 1; });
368   }
369   return finalHint;
370 };

372 Collection.prototype.findOne = function(queryObject, options, callback) {
373   var args = Array.prototype.slice.call(arguments, 0);
374   callback = args.pop();
375   queryObject = args.length ? args.shift() : {};
376   options = args.length ? args.shift() : {};  
377 
378   var fields = options.fields;
379   // Normalize fields filtering
380   if(options && options.fields){
381     fields = {};
382     if(options.fields.constructor == Array){
383       if(options.fields.length == 0) fields['_id'] = 1;
384       else for(i = 0, l = options.fields.length; i < l; i++) fields[options.fields[i]] = 1;
385     }
386     else fields = options.fields;
387   }    
388   // Unpack the options
389   var timeout  = options.timeout != null ? options.timeout : QueryCommand.OPTS_NONE;
390   var queryOptions = timeout;
391   // Build final query
392   var finalQueryObject = queryObject == null ? {} : queryObject;
393   // Validate the type of query
394   finalQueryObject = (finalQueryObject instanceof this.db.bson_serializer.ObjectID || Object.prototype.toString.call(finalQueryObject) === '[object ObjectID]') ? {'_id':finalQueryObject} : finalQueryObject;
395   // Build special selector
396   var specialSelector = {'query':finalQueryObject};
397   // Execute the command
398   var queryCommand = new QueryCommand(this.db, this.db.databaseName + "." + this.collectionName, queryOptions, 0, 1, specialSelector, fields);
399   this.db.executeCommand(queryCommand, function(err, result) {
400     if(!err && result.documents[0] && result.documents[0]['$err']) return callback(result.documents[0]['$err'], null);    
401     callback(err, result.documents[0]);
402   });

459 Collection.prototype.group = function(keys, condition, initial, reduce, command, callback) {
460   var args = Array.prototype.slice.call(arguments, 3);
461   callback = args.pop();
462   reduce = args.length ? args.shift() : null;
463   command = args.length ? args.shift() : null;
464 
465   if(command) {
466     var hash = {};
467     keys.forEach(function(key) {
468       hash[key] = 1;
469     });
470 
471     var reduceFunction = reduce != null && reduce instanceof this.db.bson_serializer.Code ? reduce : new this.db.bson_serializer.Code(reduce);
472     var selector = {'group': {
473                       'ns':this.collectionName,
474                       '$reduce': reduce,
475                       'key':hash,
476                       'cond':condition,
477                       'initial': initial}};
478 
479     this.db.executeCommand(DbCommand.createDbCommand(this.db, selector), function(err, result) {
480       var document = result.documents[0];
481       if(err == null && document.retval != null) {
482         callback(null, document.retval);
483       } else {
484         err != null ? callback(err, null) : callback(new Error("group command failed: " + document.errmsg), null);
485       }
486     });
487   } else {
488     // Create execution scope
489     var scope = reduce != null && reduce instanceof this.db.bson_serializer.Code ? reduce.scope : {};
490     // Create scope for execution
491     scope['ns'] = this.collectionName;
492     scope['keys'] = keys;
493     scope['condition'] = condition;
494     scope['initial'] = initial;
495     // Define group function
496     var groupFunction = function() {
497         var c = db[ns].find(condition);
498         var map = new Map();
499         var reduce_function = reduce;
500         while (c.hasNext()) {
501             var obj = c.next();
502 
503             var key = {};
504             for (var i = 0; i < keys.length; i++) {
505                 var k = keys[i];
506                 key[k] = obj[k];
507             }
508 
509             var aggObj = map.get(key);
510             if (aggObj == null) {
511                 var newObj = Object.extend({}, key);
512                 aggObj = Object.extend(newObj, initial);
513                 map.put(key, aggObj);
514             }
515             reduce_function(obj, aggObj);
516         }
517         return {"result": map.values()};
518       };
519 
520     // Turn function into text and replace the "result" function of the grouping function
521     var groupFunctionString = groupFunction.toString().replace(/ reduce;/, reduce.toString() + ';');
522     // Execute group
523     this.db.eval(new this.db.bson_serializer.Code(groupFunctionString, scope), function(err, results) {
524       if(err != null) {
525         callback(err, null);
526       } else {
527         if(results.constructor == Object) {
528           callback(err, results.result);
529         } else {
530           callback(err, results);
531         }
532       }
533     });
534   }
535 };
