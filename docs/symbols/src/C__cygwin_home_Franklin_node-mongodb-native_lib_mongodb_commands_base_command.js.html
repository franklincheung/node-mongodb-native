 12 BaseCommand.prototype.toBinary = function() {
 13   // Get the command op code
 14   var op_code = this.getOpCode();
 15   // Get the command data structure
 16   var command = this.getCommand();
 17   // Total Size of command
 18   var totalSize = 4*4 + command.length;
 19   // Create the command with the standard header file
 20   return BinaryParser.fromInt(totalSize) + BinaryParser.fromInt(this.requestId) + BinaryParser.fromInt(0) + BinaryParser.fromInt(op_code) + command;
 21 };

 26   return this.requestId;
