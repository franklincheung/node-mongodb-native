 15 /**
 16  * @fileoverview Defines a exports.Long class for representing a 64-bit two's-complement
 17  * integer value, which faithfully simulates the behavior of a Java "long". This
 18  * implementation is derived from exports.LongLib in GWT.
 19  *
 20  */
 21 
 22 /**
 23  * Constructs a 64-bit two's-complement integer, given its low and high 32-bit
 24  * values as *signed* integers.  See the from* functions below for more
 25  * convenient ways of constructing exports.Longs.
 26  *
 27  * The internal representation of a long is the two given signed, 32-bit values.
 28  * We use 32-bit pieces because these are the size of integers on which
 29  * Javascript performs bit-operations.  For operations like addition and
 30  * multiplication, we split each number into 16-bit pieces, which can easily be
 31  * multiplied within Javascript's floating-point representation without overflow
 32  * or change in sign.
 33  *
 34  * In the algorithms below, we frequently reduce the negative case to the
 35  * positive case by negating the input(s) and then post-processing the result.
 36  * Note that we must ALWAYS check specially whether those values are MIN_VALUE
 37  * (-2^63) because -MIN_VALUE == MIN_VALUE (since 2^63 cannot be represented as
 38  * a positive number, it overflows back into a negative).  Not handling this
 39  * case would often result in infinite recursion.
 40  *
 41  * @param {number} low  The low (signed) 32 bits of the long.
 42  * @param {number} high  The high (signed) 32 bits of the long.
 43  * @constructor
 44  */
 45 exports.Long = function(low, high) {
 46   /**
 47    * @type {number}
 48    * @private
 49    */
 50   this.low_ = low | 0;  // force into 32 signed bits.
 51 
 52   /**
 53    * @type {number}
 54    * @private
 55    */
 56   this.high_ = high | 0;  // force into 32 signed bits.
 57 };
 58

 64 /**
 65  * A cache of the exports.Long representations of small integer values.
 66  * @type {Object}
 67  * @private
 68  */
 69 exports.Long.INT_CACHE_ = {};

 72 /**
 73  * Returns a exports.Long representing the given (32-bit) integer value.
 74  * @param {number} value The 32-bit integer in question.
 75  * @return {exports.Long} The corresponding exports.Long value.
 76  */
 77 exports.Long.fromInt = function(value) {
 78   if (-128 <= value && value < 128) {
 79     var cachedObj = exports.Long.INT_CACHE_[value];
 80     if (cachedObj) {
 81       return cachedObj;
 82     }
 83   }
 84 
 85   var obj = new exports.Long(value | 0, value < 0 ? -1 : 0);
 86   if (-128 <= value && value < 128) {
 87     exports.Long.INT_CACHE_[value] = obj;
 88   }
 89   return obj;
 90 };

 93 /**
 94  * Returns a exports.Long representing the given value, provided that it is a finite
 95  * number.  Otherwise, zero is returned.
 96  * @param {number} value The number in question.
 97  * @return {exports.Long} The corresponding exports.Long value.
 98  */
 99 exports.Long.fromNumber = function(value) {
100   if (isNaN(value) || !isFinite(value)) {
101     return exports.Long.ZERO;
102   } else if (value <= -exports.Long.TWO_PWR_63_DBL_) {
103     return exports.Long.MIN_VALUE;
104   } else if (value + 1 >= exports.Long.TWO_PWR_63_DBL_) {
105     return exports.Long.MAX_VALUE;
106   } else if (value < 0) {
107     return exports.Long.fromNumber(-value).negate();
108   } else {
109     return new exports.Long(
110                (value % exports.Long.TWO_PWR_32_DBL_) | 0,
111                (value / exports.Long.TWO_PWR_32_DBL_) | 0);
112   }
113 };

117  * Returns a exports.Long representing the 64-bit integer that comes by concatenating
118  * the given high and low bits.  Each is assumed to use 32 bits.

121  * @return {exports.Long} The corresponding exports.Long value.
122  */
123 exports.Long.fromBits = function(lowBits, highBits) {

128 /**
129  * Returns a exports.Long representation of the given string, written using the given
130  * radix.
131  * @param {string} str The textual representation of the exports.Long.
132  * @param {number} opt_radix The radix in which the text is written.
133  * @return {exports.Long} The corresponding exports.Long value.
134  */
135 exports.Long.fromString = function(str, opt_radix) {
136   if (str.length == 0) {
137     throw Error('number format error: empty string');
138   }
139 
140   var radix = opt_radix || 10;
141   if (radix < 2 || 36 < radix) {
142     throw Error('radix out of range: ' + radix);
143   }
144 
145   if (str.charAt(0) == '-') {
146     return exports.Long.fromString(str.substring(1), radix).negate();
147   } else if (str.indexOf('-') >= 0) {
148     throw Error('number format error: interior "-" character: ' + str);
149   }
150 
151   // Do several (8) digits each time through the loop, so as to
152   // minimize the calls to the very expensive emulated div.
153   var radixToPower = exports.Long.fromNumber(Math.pow(radix, 8));
154 
155   var result = exports.Long.ZERO;
156   for (var i = 0; i < str.length; i += 8) {
157     var size = Math.min(8, str.length - i);
158     var value = parseInt(str.substring(i, i + size), radix);
159     if (size < 8) {
160       var power = exports.Long.fromNumber(Math.pow(radix, size));
161       result = result.multiply(power).add(exports.Long.fromNumber(value));
162     } else {
163       result = result.multiply(radixToPower);
164       result = result.add(exports.Long.fromNumber(value));
165     }
166   }
167   return result;
168 };

175 /**
176  * Number used repeated below in calculations.  This must appear before the
177  * first call to any from* function below.
178  * @type {number}
179  * @private
180  */
181 exports.Long.TWO_PWR_16_DBL_ = 1 << 16;
182 
183 /**
184  * @type {number}
185  * @private
186  */
187 exports.Long.TWO_PWR_24_DBL_ = 1 << 24;
188 
189 /**
190  * @type {number}
191  * @private
192  */
193 exports.Long.TWO_PWR_32_DBL_ =
194     exports.Long.TWO_PWR_16_DBL_ * exports.Long.TWO_PWR_16_DBL_;
195 
196 /**
197  * @type {number}
198  * @private
199  */
200 exports.Long.TWO_PWR_31_DBL_ =
201     exports.Long.TWO_PWR_32_DBL_ / 2;
202 
203 /**
204  * @type {number}
205  * @private
206  */
207 exports.Long.TWO_PWR_48_DBL_ =
208     exports.Long.TWO_PWR_32_DBL_ * exports.Long.TWO_PWR_16_DBL_;
209 
210 /**
211  * @type {number}
212  * @private
213  */
214 exports.Long.TWO_PWR_64_DBL_ =
215     exports.Long.TWO_PWR_32_DBL_ * exports.Long.TWO_PWR_32_DBL_;
216 
217 /**
218  * @type {number}
219  * @private
220  */
221 exports.Long.TWO_PWR_63_DBL_ =
222     exports.Long.TWO_PWR_64_DBL_ / 2;

226 exports.Long.ZERO = exports.Long.fromInt(0);
227 
228 /** @type {exports.Long} */
229 exports.Long.ONE = exports.Long.fromInt(1);

232 exports.Long.NEG_ONE = exports.Long.fromInt(-1);

262 /**
263  * @param {number} opt_radix The radix in which the text should be written.
264  * @return {string} The textual representation of this value.
265  */
266 exports.Long.prototype.toString = function(opt_radix) {
267   var radix = opt_radix || 10;
268   if (radix < 2 || 36 < radix) {
269     throw Error('radix out of range: ' + radix);
270   }
271 
272   if (this.isZero()) {
273     return '0';
274   }
275 
276   if (this.isNegative()) {
277     if (this.equals(exports.Long.MIN_VALUE)) {
278       // We need to change the exports.Long value before it can be negated, so we remove
279       // the bottom-most digit in this base and then recurse to do the rest.
280       var radixLong = exports.Long.fromNumber(radix);
281       var div = this.div(radixLong);
282       var rem = div.multiply(radixLong).subtract(this);
283       return div.toString(radix) + rem.toInt().toString(radix);
284     } else {
285       return '-' + this.negate().toString(radix);
286     }
287   }
288 
289   // Do several (6) digits each time through the loop, so as to
290   // minimize the calls to the very expensive emulated div.
291   var radixToPower = exports.Long.fromNumber(Math.pow(radix, 6));
292 
293   var rem = this;
294   var result = '';
295   while (true) {
296     var remDiv = rem.div(radixToPower);
297     var intval = rem.subtract(remDiv.multiply(radixToPower)).toInt();
298     var digits = intval.toString(radix);
299 
300     rem = remDiv;
301     if (rem.isZero()) {
302       return digits + result;
303     } else {
304       while (digits.length < 6) {
305         digits = '0' + digits;
306       }
307       result = '' + digits + result;
308     }
309   }
310 };

332 /**
333  * @return {number} Returns the number of bits needed to represent the absolute
334  *     value of this exports.Long.
335  */
336 exports.Long.prototype.getNumBitsAbs = function() {
337   if (this.isNegative()) {
338     if (this.equals(exports.Long.MIN_VALUE)) {
339       return 64;
340     } else {
341       return this.negate().getNumBitsAbs();
342     }
343   } else {
344     var val = this.high_ != 0 ? this.high_ : this.low_;
345     for (var bit = 31; bit > 0; bit--) {
346       if ((val & (1 << bit)) != 0) {
347         break;
348       }
349     }
350     return this.high_ != 0 ? bit + 33 : bit + 1;
351   }
352 };

427 /**
428  * Compares this exports.Long with the given one.
429  * @param {exports.Long} other exports.Long to compare against.
430  * @return {number} 0 if they are the same, 1 if the this is greater, and -1
431  *     if the given one is greater.
432  */
433 exports.Long.prototype.compare = function(other) {
434   if (this.equals(other)) {
435     return 0;
436   }
437 
438   var thisNeg = this.isNegative();
439   var otherNeg = other.isNegative();
440   if (thisNeg && !otherNeg) {
441     return -1;
442   }
443   if (!thisNeg && otherNeg) {
444     return 1;
445   }
446 
447   // at this point, the signs are the same, so subtraction will not overflow
448   if (this.subtract(other).isNegative()) {
449     return -1;
450   } else {
451     return 1;
452   }
453 };

456 /** @return {exports.Long} The negation of this value. */
457 exports.Long.prototype.negate = function() {
458   if (this.equals(exports.Long.MIN_VALUE)) {
459     return exports.Long.MIN_VALUE;
460   } else {
461     return this.not().add(exports.Long.ONE);
462   }
463 };

466 /**
467  * Returns the sum of this and the given exports.Long.
468  * @param {exports.Long} other exports.Long to add to this one.
469  * @return {exports.Long} The sum of this and the given exports.Long.
470  */
471 exports.Long.prototype.add = function(other) {
472   // Divide each number into 4 chunks of 16 bits, and then sum the chunks.
473 
474   var a48 = this.high_ >>> 16;
475   var a32 = this.high_ & 0xFFFF;
476   var a16 = this.low_ >>> 16;
477   var a00 = this.low_ & 0xFFFF;
478 
479   var b48 = other.high_ >>> 16;
480   var b32 = other.high_ & 0xFFFF;
481   var b16 = other.low_ >>> 16;
482   var b00 = other.low_ & 0xFFFF;
483 
484   var c48 = 0, c32 = 0, c16 = 0, c00 = 0;
485   c00 += a00 + b00;
486   c16 += c00 >>> 16;
487   c00 &= 0xFFFF;
488   c16 += a16 + b16;
489   c32 += c16 >>> 16;
490   c16 &= 0xFFFF;
491   c32 += a32 + b32;
492   c48 += c32 >>> 16;
493   c32 &= 0xFFFF;
494   c48 += a48 + b48;
495   c48 &= 0xFFFF;
496   return exports.Long.fromBits((c16 << 16) | c00, (c48 << 16) | c32);
497 };

510 /**
511  * Returns the product of this and the given long.
512  * @param {exports.Long} other exports.Long to multiply with this.
513  * @return {exports.Long} The product of this and the other.
514  */
515 exports.Long.prototype.multiply = function(other) {
516   if (this.isZero()) {
517     return exports.Long.ZERO;
518   } else if (other.isZero()) {
519     return exports.Long.ZERO;
520   }
521 
522   if (this.equals(exports.Long.MIN_VALUE)) {
523     return other.isOdd() ? exports.Long.MIN_VALUE : exports.Long.ZERO;
524   } else if (other.equals(exports.Long.MIN_VALUE)) {
525     return this.isOdd() ? exports.Long.MIN_VALUE : exports.Long.ZERO;
526   }
527 
528   if (this.isNegative()) {
529     if (other.isNegative()) {
530       return this.negate().multiply(other.negate());
531     } else {
532       return this.negate().multiply(other).negate();
533     }
534   } else if (other.isNegative()) {
535     return this.multiply(other.negate()).negate();
536   }
537 
538   // If both longs are small, use float multiplication
539   if (this.lessThan(exports.Long.TWO_PWR_24_) &&
540       other.lessThan(exports.Long.TWO_PWR_24_)) {
541     return exports.Long.fromNumber(this.toNumber() * other.toNumber());
542   }
543 
544   // Divide each long into 4 chunks of 16 bits, and then add up 4x4 products.
545   // We can skip products that would overflow.
546 
547   var a48 = this.high_ >>> 16;
548   var a32 = this.high_ & 0xFFFF;
549   var a16 = this.low_ >>> 16;
550   var a00 = this.low_ & 0xFFFF;
551 
552   var b48 = other.high_ >>> 16;
553   var b32 = other.high_ & 0xFFFF;
554   var b16 = other.low_ >>> 16;
555   var b00 = other.low_ & 0xFFFF;
556 
557   var c48 = 0, c32 = 0, c16 = 0, c00 = 0;
558   c00 += a00 * b00;
559   c16 += c00 >>> 16;
560   c00 &= 0xFFFF;
561   c16 += a16 * b00;
562   c32 += c16 >>> 16;
563   c16 &= 0xFFFF;
564   c16 += a00 * b16;
565   c32 += c16 >>> 16;
566   c16 &= 0xFFFF;
567   c32 += a32 * b00;
568   c48 += c32 >>> 16;
569   c32 &= 0xFFFF;
570   c32 += a16 * b16;
571   c48 += c32 >>> 16;
572   c32 &= 0xFFFF;
573   c32 += a00 * b32;
574   c48 += c32 >>> 16;
575   c32 &= 0xFFFF;
576   c48 += a48 * b00 + a32 * b16 + a16 * b32 + a00 * b48;
577   c48 &= 0xFFFF;
578   return exports.Long.fromBits((c16 << 16) | c00, (c48 << 16) | c32);
579 };

582 /**
583  * Returns this exports.Long divided by the given one.
584  * @param {exports.Long} other exports.Long by which to divide.
585  * @return {exports.Long} This exports.Long divided by the given one.
586  */
587 exports.Long.prototype.div = function(other) {
588   if (other.isZero()) {
589     throw Error('division by zero');
590   } else if (this.isZero()) {
591     return exports.Long.ZERO;
592   }
593 
594   if (this.equals(exports.Long.MIN_VALUE)) {
595     if (other.equals(exports.Long.ONE) ||
596         other.equals(exports.Long.NEG_ONE)) {
597       return exports.Long.MIN_VALUE;  // recall that -MIN_VALUE == MIN_VALUE
598     } else if (other.equals(exports.Long.MIN_VALUE)) {
599       return exports.Long.ONE;
600     } else {
601       // At this point, we have |other| >= 2, so |this/other| < |MIN_VALUE|.
602       var halfThis = this.shiftRight(1);
603       var approx = halfThis.div(other).shiftLeft(1);
604       if (approx.equals(exports.Long.ZERO)) {
605         return other.isNegative() ? exports.Long.ONE : exports.Long.NEG_ONE;
606       } else {
607         var rem = this.subtract(other.multiply(approx));
608         var result = approx.add(rem.div(other));
609         return result;
610       }
611     }
612   } else if (other.equals(exports.Long.MIN_VALUE)) {
613     return exports.Long.ZERO;
614   }
615 
616   if (this.isNegative()) {
617     if (other.isNegative()) {
618       return this.negate().div(other.negate());
619     } else {
620       return this.negate().div(other).negate();
621     }
622   } else if (other.isNegative()) {
623     return this.div(other.negate()).negate();
624   }
625 
626   // Repeat the following until the remainder is less than other:  find a
627   // floating-point that approximates remainder / other *from below*, add this
628   // into the result, and subtract it from the remainder.  It is critical that
629   // the approximate value is less than or equal to the real value so that the
630   // remainder never becomes negative.
631   var res = exports.Long.ZERO;
632   var rem = this;
633   while (rem.greaterThanOrEqual(other)) {
634     // Approximate the result of division. This may be a little greater or
635     // smaller than the actual value.
636     var approx = Math.max(1, Math.floor(rem.toNumber() / other.toNumber()));
637 
638     // We will tweak the approximate result by changing it in the 48-th digit or
639     // the smallest non-fractional digit, whichever is larger.
640     var log2 = Math.ceil(Math.log(approx) / Math.LN2);
641     var delta = (log2 <= 48) ? 1 : Math.pow(2, log2 - 48);
642 
643     // Decrease the approximation until it is smaller than the remainder.  Note
644     // that if it is too large, the product overflows and is negative.
645     var approxRes = exports.Long.fromNumber(approx);
646     var approxRem = approxRes.multiply(other);
647     while (approxRem.isNegative() || approxRem.greaterThan(rem)) {
648       approx -= delta;
649       approxRes = exports.Long.fromNumber(approx);
650       approxRem = approxRes.multiply(other);
651     }
652 
653     // We know the answer can't be zero... and actually, zero would cause
654     // infinite recursion since we would make no progress.
655     if (approxRes.isZero()) {
656       approxRes = exports.Long.ONE;
657     }
658 
659     res = res.add(approxRes);
660     rem = rem.subtract(approxRem);
661   }
662   return res;
663 };

682 /**
683  * Returns the bitwise-AND of this exports.Long and the given one.
684  * @param {exports.Long} other The exports.Long with which to AND.
685  * @return {exports.Long} The bitwise-AND of this and the other.

687 exports.Long.prototype.and = function(other) {

717  * @param {number} numBits The number of bits by which to shift.
718  * @return {exports.Long} This shifted to the left by the given amount.
719  */
720 exports.Long.prototype.shiftLeft = function(numBits) {
721   numBits &= 63;
722   if (numBits == 0) {
723     return this;
724   } else {
725     var low = this.low_;
726     if (numBits < 32) {
727       var high = this.high_;
728       return exports.Long.fromBits(
729                  low << numBits,
730                  (high << numBits) | (low >>> (32 - numBits)));
731     } else {
732       return exports.Long.fromBits(0, low << (numBits - 32));
733     }
734   }
