1 var BinaryParser = require('../bson/binary_parser').BinaryParser,
  2   Integer = require('../goog/math/integer').Integer,
  3   Long = require('../goog/math/long').Long;
  4 
  5 /**
  6   Reply message from mongo db
  7 **/
  8 var MongoReply = exports.MongoReply = function(db, binary_reply) {
  9   this.documents = [];
 10   var index = 0;
 11   // Unpack the standard header first
 12   var messageLength = BinaryParser.toInt(binary_reply.substr(index, 4));
 13   index = index + 4;
 14   // Fetch the request id for this reply
 15   this.requestId = BinaryParser.toInt(binary_reply.substr(index, 4));
 16   index = index + 4;
 17   // Fetch the id of the request that triggered the response
 18   this.responseTo = BinaryParser.toInt(binary_reply.substr(index, 4));
 19   // Skip op-code field
 20   index = index + 4 + 4;
 21   // Unpack the reply message
 22   this.responseFlag = BinaryParser.toInt(binary_reply.substr(index, 4));
 23   index = index + 4;  
 24   // Unpack the cursor id (a 64 bit long integer)
 25   this.cursorId = new db.bson_serializer.BSON.toLong(BinaryParser.toInt(binary_reply.substr(index, 4)), BinaryParser.toInt(binary_reply.substr(index + 4, 4)));
 26   index = index + 8;
 27   // Unpack the starting from
 28   this.startingFrom = BinaryParser.toInt(binary_reply.substr(index, 4));
 29   index = index + 4;
 30   // Unpack the number of objects returned
 31   this.numberReturned = BinaryParser.toInt(binary_reply.substr(index, 4));
 32   index = index + 4;
 33   // Let's unpack all the bson document, deserialize them and store them
 34   for(var object_index = 0; object_index < this.numberReturned; object_index++) {
 35     // Read the size of the bson object
 36     var bsonObjectSize = BinaryParser.toInt(binary_reply.substr(index, 4));
 37     
 38     // sys.debug("--------------------------------------------------- incoming")
 39     // BinaryParser.hprint(binary_reply.substr(index, bsonObjectSize))
 40     
 41     // Read the entire object and deserialize it
 42     this.documents.push(db.bson_deserializer.BSON.deserialize(binary_reply.substr(index, bsonObjectSize)));
 43     // Adjust for next object
 44     index = index + bsonObjectSize;
 45   }
 46 };
 47 
 48 MongoReply.prototype.is_error = function(){
 49   if(this.documents.length == 1) {
 50     return this.documents[0].ok == 1 ? false : true;
 51   }
 52   return false;
 53 };
 54 
 55 MongoReply.prototype.error_message = function() {
 56   return this.documents.length == 1 && this.documents[0].ok == 1 ? '' : this.documents[0].errmsg;
 57 };
