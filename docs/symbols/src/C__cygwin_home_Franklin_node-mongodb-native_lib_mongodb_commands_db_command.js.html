  6 /**
  7 Represents a command for the database
  8 @constructor
  9 @augments BaseCommand
 10 */
 11 var DbCommand = exports.DbCommand = function(db, collectionName, queryOptions, numberToSkip, numberToReturn, query, returnFieldSelector) {
 12   QueryCommand.call(db, this);
 13 
 14   this.collectionName = collectionName;
 15   this.queryOptions = queryOptions;
 16   this.numberToSkip = numberToSkip;
 17   this.numberToReturn = numberToReturn;
 18   this.query = query;
 19   this.returnFieldSelector = returnFieldSelector;
 20   this.db = db;
 21 };

 51 DbCommand.createAuthenticationCommand = function(db, username, password, nonce) {
 52   // Generate keys used for authentication
 53   var hash_password = MD5.hex_md5(username + ":mongo:" + password);
 54   var key = MD5.hex_md5(nonce + username + hash_password);
 55   var selector = {'authenticate':1, 'user':username, 'nonce':nonce, 'key':key};
 56   // Create db command
 57   return new DbCommand(db, db.databaseName + "." + DbCommand.SYSTEM_COMMAND_COLLECTION, QueryCommand.OPTS_NO_CURSOR_TIMEOUT, 0, -1, selector, null);
 58 };

 64 DbCommand.createCreateCollectionCommand = function(db, collectionName, options) {
 65   var selector = {'create':collectionName};
 66   // Modify the options to ensure correct behaviour
 67   for(var name in options) {
 68     if(options[name] != null && options[name].constructor != Function) selector[name] = options[name];
 69   }
 70   // Execute the command
 71   return new DbCommand(db, db.databaseName + "." + DbCommand.SYSTEM_COMMAND_COLLECTION, QueryCommand.OPTS_NO_CURSOR_TIMEOUT, 0, -1, selector, null);
 72 };

100 DbCommand.createCreateIndexCommand = function(db, collectionName, fieldOrSpec, unique) {
101   var finalUnique = unique == null ? false : unique;
102   var fieldHash = {};
103   var finalFieldOrSpec = fieldOrSpec.constructor == String ? [[fieldOrSpec, 1]] : fieldOrSpec;
104   var indexes = [];
105 
106   // Get all the fields
107   finalFieldOrSpec.forEach(function(indexArray) {
108     var indexArrayFinal = indexArray;
109     if(indexArrayFinal.length == 1) indexArrayFinal[1] = 1;
110     fieldHash[indexArrayFinal[0]] = indexArrayFinal[1];
111     indexes.push(indexArrayFinal[0] + "_" + indexArrayFinal[1]);
112   });
113   // Generate the index name
114   var indexName = indexes.join("_");
115   // Build the selector
116   var selector = {'ns':(db.databaseName + "." + collectionName), 'unique':finalUnique, 'key':fieldHash, 'name':indexName};
117   // Create the insert command for the index and return the document
118   return new InsertCommand(db, db.databaseName + "." + DbCommand.SYSTEM_INDEX_COLLECTION, false).add(selector);
119 };
