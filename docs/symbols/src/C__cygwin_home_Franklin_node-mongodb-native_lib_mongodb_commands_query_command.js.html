1 var BaseCommand = require('./base_command').BaseCommand,
  2   BinaryParser = require('../bson/binary_parser').BinaryParser,
  3   inherits = require('sys').inherits;
  4 
  5 /**
  6 Object which represents a query command to the database  
  7 @constructor
  8 @param db Object representing the database
  9 @param collectionName String representing the name of the collection to query
 10 **/
 11 var QueryCommand = exports.QueryCommand = function(db, collectionName, queryOptions, numberToSkip, numberToReturn, query, returnFieldSelector) {
 12   BaseCommand.call(this);
 13   
 14   this.collectionName = collectionName;
 15   this.queryOptions = queryOptions;
 16   this.numberToSkip = numberToSkip;
 17   this.numberToReturn = numberToReturn;
 18   this.query = query;
 19   this.returnFieldSelector = returnFieldSelector;
 20   this.db = db;
 21 };
 22 
 23 inherits(QueryCommand, BaseCommand);
 24 /**
 25 Returns the op code
 26 @function
 27 */
 28 QueryCommand.prototype.getOpCode = function() {
 29   return BaseCommand.OP_QUERY;
 30 };
 31 
 32 /*
 33 struct {
 34     MsgHeader header;                 // standard message header
 35     int32     opts;                   // query options.  See below for details.
 36     cstring   fullCollectionName;     // "dbname.collectionname"
 37     int32     numberToSkip;           // number of documents to skip when returning results
 38     int32     numberToReturn;         // number of documents to return in the first OP_REPLY
 39     BSON      query ;                 // query object.  See below for details.
 40   [ BSON      returnFieldSelector; ]  // OPTIONAL : selector indicating the fields to return.  See below for details.
 41 }
 42 */
 43 /**
 44 Generates and returns the command string
 45 @function
 46 */
 47 QueryCommand.prototype.getCommand = function() {
 48   // Generate the command string
 49   var command_string = BinaryParser.fromInt(this.queryOptions) + BinaryParser.encode_cstring(this.collectionName);
 50   command_string = command_string + BinaryParser.fromInt(this.numberToSkip) + BinaryParser.fromInt(this.numberToReturn);
 51   command_string = command_string + this.db.bson_serializer.BSON.serialize(this.query);
 52   if(this.returnFieldSelector != null)  {
 53     var count = 0; for(var name in this.returnFieldSelector) { count += 1; }
 54     if(count > 0) command_string = command_string + this.db.bson_serializer.BSON.serialize(this.returnFieldSelector);
 55   }
 56   return command_string;
 57 };
