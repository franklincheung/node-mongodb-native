 49 BSON.deserialize = function(data, is_array_item, returnData, returnArray) {
 50   // sys.debug("====================================== JSJS deserialize")
 51   // The return data
 52   var return_data = {};
 53   var return_array = [];
 54   // Index of decoding in the binary file
 55   var index = 0;
 56   // Split of the first 4 characters to get the number of bytes
 57   var size = BinaryParser.toInt(data.substr(0, 4));
 58   // Adjust index
 59   index = index + 4;
 60   
 61   while(index < data.length) {
 62     // Read the first byte indicating the type of object
 63     var type = BinaryParser.toSmall(data.substr(index, 1));
 64     var insert_index = 0;
 65     // Adjust for the type of element
 66     index = index + 1;
 67     // If it's a string decode the value
 68     if(type == BSON.BSON_DATA_STRING) {
 69       // Read the null terminated string (indexof until first 0)
 70       var string_end_index = data.indexOf('\0', index);
 71       var string_name = data.substring(index, string_end_index);
 72       // Ajust index to point to the end of the string
 73       index = string_end_index + 1;
 74       // If we have an index read the array index      
 75       if(is_array_item) insert_index = parseInt(string_name, 10);
 76 
 77       // Read the length of the string (next 4 bytes)
 78       var string_size = BinaryParser.toInt(data.substr(index, 4));
 79       // Adjust the index to point at start of string
 80       index = index + 4;      
 81       // Read the string
 82       var value = BinaryParser.decode_utf8(data.substr(index, string_size - 1));
 83       // Adjust the index with the size of the string
 84       index = index + string_size;
 85       // Set the data on the object
 86       is_array_item ? return_array[insert_index] = value : return_data[string_name] = value;
 87     } else if(type == BSON.BSON_DATA_NULL) {
 88       // Read the null terminated string (indexof until first 0)
 89       var string_end_index = data.indexOf('\0', index);
 90       var string_name = data.substring(index, string_end_index);
 91       // Ajust index to point to the end of the string
 92       index = string_end_index + 1;
 93       // If we have an index read the array index      
 94       if(is_array_item) insert_index = parseInt(string_name, 10);
 95       // Set the data on the object
 96       is_array_item ? return_array[insert_index] = null : return_data[string_name] = null;
 97     } else if(type == BSON.BSON_DATA_INT) {
 98       // Read the null terminated string (indexof until first 0)
 99       var string_end_index = data.indexOf('\0', index);
100       var string_name = data.substring(index, string_end_index);
101       // Ajust index to point to the end of the string
102       index = string_end_index + 1;
103       // If we have an index read the array index      
104       if(is_array_item) insert_index = parseInt(string_name, 10);
105       // Read the number value
106       var value = BinaryParser.toInt(data.substr(index, 4));
107       // Adjust the index with the size
108       index = index + 4;
109       // Set the data on the object
110       is_array_item ? return_array[insert_index] = value : return_data[string_name] = value;
111     } else if(type == BSON.BSON_DATA_LONG) {
112       // Read the null terminated string (indexof until first 0)
113       var string_end_index = data.indexOf('\0', index);
114       var string_name = data.substring(index, string_end_index);
115       // Ajust index to point to the end of the string
116       index = string_end_index + 1;
117       // If we have an index read the array index      
118       if(is_array_item) insert_index = parseInt(string_name, 10);
119       // Read the number value
120       var low_bits = Integer.fromInt(BinaryParser.toInt(data.substr(index, 4)));
121       var high_bits = Integer.fromInt(BinaryParser.toInt(data.substr(index + 4, 4)));
122       // Create to integers
123       var value = new Long(low_bits, high_bits);
124       // Adjust the index with the size
125       index = index + 8;
126       // Set the data on the object
127       is_array_item ? return_array[insert_index] = value : return_data[string_name] = value;
128     } else if(type == BSON.BSON_DATA_NUMBER) {
129       // Read the null terminated string (indexof until first 0)
130       var string_end_index = data.indexOf('\0', index);
131       var string_name = data.substring(index, string_end_index);
132       // Ajust index to point to the end of the string
133       index = string_end_index + 1;
134       // If we have an index read the array index      
135       if(is_array_item) insert_index = parseInt(string_name, 10);
136       // Read the number value
137       var value = BinaryParser.toDouble(data.substr(index, 8));
138       // Adjust the index with the size
139       index = index + 8;
140       // Set the data on the object
141       is_array_item ? return_array[insert_index] = value : return_data[string_name] = value;
142     } else if(type == BSON.BSON_DATA_OBJECT) {
143       // Read the null terminated string (indexof until first 0)
144       var string_end_index = data.indexOf('\0', index);
145       var string_name = data.substring(index, string_end_index);
146       // Ajust index to point to the end of the string
147       index = string_end_index + 1;      
148       // If we have an index read the array index      
149       if(is_array_item) insert_index = parseInt(string_name, 10);
150       // Read the size of the object
151       var object_size = BinaryParser.toInt(data.substr(index, 4));
152       // Do a substr based on the size and parse the sub object
153       var object_data = data.substr(index, object_size);
154       // Parse the object
155       var value = BSON.deserialize(object_data, false);
156       // Adjust the index for the next value
157       index = index + object_size;
158       // Set the data on the object
159       is_array_item ? return_array[insert_index] = value : return_data[string_name] = value;
160     } else if(type == BSON.BSON_DATA_ARRAY) {
161       // Read the null terminated string (indexof until first 0)
162       var string_end_index = data.indexOf('\0', index);
163       var string_name = data.substring(index, string_end_index);
164       // Ajust index to point to the end of the string
165       index = string_end_index + 1;
166       // If we have an index read the array index      
167       if(is_array_item) insert_index = parseInt(string_name, 10);
168       // Read the size of the object
169       var array_size = BinaryParser.toInt(data.substr(index, 4));
170       // Let's split off the data and parse all elements (keeping in mind the elements)
171       var array_data = data.substr(index, array_size);
172       // Parse the object
173       var value = BSON.deserialize(array_data, true);
174       // Adjust the index for the next value
175       index = index + array_size;
176       // Set the data on the object
177       is_array_item ? return_array[insert_index] = value : return_data[string_name] = value;
178     } else if(type == BSON.BSON_DATA_BOOLEAN) {
179       // Read the null terminated string (indexof until first 0)
180       var string_end_index = data.indexOf('\0', index);
181       var string_name = data.substring(index, string_end_index);
182       // Ajust index to point to the end of the string
183       index = string_end_index + 1;
184       // If we have an index read the array index      
185       if(is_array_item) insert_index = parseInt(string_name, 10);
186       // Read the length of the string (next 4 bytes)
187       var boolean_value = BinaryParser.toSmall(data.substr(index, 1));
188       var value = boolean_value == 1 ? true : false;
189       // Adjust the index
190       index = index + 1;
191       // Set the data on the object
192       is_array_item ? return_array[insert_index] = value : return_data[string_name] = value;
193     } else if(type == BSON.BSON_DATA_DATE) {
194       // Read the null terminated string (indexof until first 0)
195       var string_end_index = data.indexOf('\0', index);
196       var string_name = data.substring(index, string_end_index);
197       // Ajust index to point to the end of the string
198       index = string_end_index + 1;
199       // If we have an index read the array index      
200       if(is_array_item) insert_index = parseInt(string_name, 10);
201       // Read the number value
202       var low_bits = Integer.fromInt(BinaryParser.toInt(data.substr(index, 4)));
203       var high_bits = Integer.fromInt(BinaryParser.toInt(data.substr(index + 4, 4)));
204       // Create to integers
205       var value_in_seconds = new Long(low_bits, high_bits).toNumber();
206       // Calculate date with miliseconds
207       var value = new Date();
208       value.setTime(value_in_seconds);
209       // Adjust the index
210       index = index + 8;
211       // Set the data on the object
212       is_array_item ? return_array[insert_index] = value : return_data[string_name] = value;
213     } else if(type == BSON.BSON_DATA_OID) {
214       // Read the null terminated string (indexof until first 0)
215       var string_end_index = data.indexOf('\0', index);
216       var string_name = data.substring(index, string_end_index);
217       // Ajust index to point to the end of the string
218       index = string_end_index + 1;
219       // If we have an index read the array index      
220       if(is_array_item) insert_index = parseInt(string_name, 10);
221       // Read the oid (12 bytes)
222       var oid = data.substr(index, 12);
223       // Calculate date with miliseconds
224       var value = new exports.ObjectID(oid);
225       // Adjust the index
226       index = index + 12;
227       // Set the data on the object
228       is_array_item ? return_array[insert_index] = value : return_data[string_name] = value;
229     } else if(type == BSON.BSON_DATA_CODE_W_SCOPE) {
230       // Read the null terminated string (indexof until first 0)
231       var string_end_index = data.indexOf('\0', index);
232       var string_name = data.substring(index, string_end_index);
233       // Ajust index to point to the end of the string
234       index = string_end_index + 1;
235       // If we have an index read the array index      
236       if(is_array_item) insert_index = parseInt(string_name, 10);
237       // Unpack the integer sizes
238       var total_code_size = BinaryParser.toInt(data.substr(index, 4));
239       index = index + 4;
240       var string_size = BinaryParser.toInt(data.substr(index, 4));
241       index = index + 4;
242       // Read the string + terminating null
243       var code_string = BinaryParser.decode_utf8(data.substr(index, string_size - 1));
244       index = index + string_size;
245       // Get the bson object
246       var bson_object_size = total_code_size - string_size - 8;
247       var bson_object_string = data.substr(index, bson_object_size);
248       index = index + bson_object_size;
249       // Parse the bson object
250       var scope_object = BSON.deserialize(bson_object_string, false);
251       // Create code object
252       var value = new exports.Code(code_string, scope_object);
253       // Set the data on the object
254       is_array_item ? return_array[insert_index] = value : return_data[string_name] = value;
255     } else if(type == BSON.BSON_DATA_REGEXP) {
256       // Read the null terminated string (indexof until first 0)
257       var string_end_index = data.indexOf('\0', index);
258       var string_name = data.substring(index, string_end_index);
259       // Ajust index to point to the end of the string
260       index = string_end_index + 1;
261       // If we have an index read the array index      
262       if(is_array_item) insert_index = parseInt(string_name, 10);
263       // Read characters until end of regular expression
264       var reg_exp_array = [];
265       var chr = 1;
266       while(BinaryParser.toByte((chr = data.charAt(index++))) != 0) {
267         reg_exp_array.push(chr);
268       }
269       // Read the options for the regular expression
270       var options_array = [];
271       while(BinaryParser.toByte((chr = data.charAt(index++))) != 0) {
272         options_array.push(chr);
273       }
274       // Regular expression
275       var value = new RegExp(reg_exp_array.join(''), options_array.join(''));
276       // Set the data on the object
277       is_array_item ? return_array[insert_index] = value : return_data[string_name] = value;
278     } else if(type == BSON.BSON_DATA_BINARY) {
279       // Read the null terminated string (indexof until first 0)
280       var string_end_index = data.indexOf('\0', index);
281       var string_name = data.substring(index, string_end_index);
282       // Ajust index to point to the end of the string
283       index = string_end_index + 1;
284       // If we have an index read the array index      
285       if(is_array_item) insert_index = parseInt(string_name, 10);      
286       // The total number of bytes after subtype
287       var total_number_of_bytes = BinaryParser.toInt(data.substr(index, 4));
288       index = index + 4;
289       // Decode the subtype
290       var sub_type = BinaryParser.toByte(data.substr(index, 1));
291       index = index + 1;
292       // Binary object
293       var value = new exports.Binary();
294       value.sub_type = sub_type;
295       // Read the size of the binary
296       var number_of_bytes = BinaryParser.toInt(data.substr(index, 4));
297       index = index + 4;
298       // Read the next bytes into our Binary object
299       var bin_data = data.substr(index, number_of_bytes);
300       value.write(bin_data);
301       // Adjust index with number of bytes
302       index = index + number_of_bytes;
303       // Set the data on the object
304       is_array_item ? return_array[insert_index] = value : return_data[string_name] = value;
305     }
306   }
307   // Check if we have a db reference
308   if(!is_array_item && return_data['$ref'] != null) {
309     return_data = new exports.DBRef(return_data['$ref'], return_data['$id'], return_data['$db']);
310   }
311 
312   // Return the data
313   return is_array_item ? return_array : return_data;
314 };

347   var encoded_string = '';

353   }
354   // Return the encoded string
355   return encoded_string;

371 BSON.encodeCode = function(code, checkKeys) {
372   // Get the code_string
373   var code_string = BSON.encodeString(code.code);
374   var scope = BinaryParser.fromInt(5) + BinaryParser.fromByte(0);
375   // Encode the scope (a hash of values or ordered hash)
376   if(code.scope != null) {
377     scope = BSON.encodeValue('', null, code.scope, false, checkKeys);
378     scope = scope.substring(1, scope.length);
379   }
380   // Calculate lengths
381   var total_length = code_string.length - 4 + scope.length + 8;
382   return BinaryParser.fromInt(total_length) + code_string + scope;
383 };

385 BSON.encodeRegExp = function(regexp) {
386   // Get regular expression
387   var clean_regexp = regexp.toString().match(/\/.*\//, '');
388   clean_regexp = clean_regexp[0].substring(1, clean_regexp[0].length - 1);
389   var options = regexp.toString().substr(clean_regexp.length + 2);
390   var options_array = [];
391   // Extract all options that are legal and sort them alphabetically
392   for(var index = 0; index < options.length; index++) {
393     var chr = options.charAt(index);
394     if(chr == 'i' || chr == 'm' || chr == 'x') options_array.push(chr);
395   }
396   // Don't need to sort the alphabetically as it's already done by javascript on creation of a Regexp obejct
397   options = options_array.join('');
398   // Encode the regular expression
399   return BinaryParser.encode_cstring(clean_regexp) + BinaryParser.encode_cstring(options);
400 };

432 BSON.encodeValue = function(encoded_string, variable, value, top_level, checkKeys) {
433   var variable_encoded = variable == null ? '' : BinaryParser.encode_cstring(variable);
434   if(checkKeys && variable != null)BSON.checkKey(variable);
435 
436   if(value == null) {
437     encoded_string += BinaryParser.fromByte(BSON.BSON_DATA_NULL) + variable_encoded;
438   } else if(value.constructor == String) {
439     encoded_string += BinaryParser.fromByte(BSON.BSON_DATA_STRING) + variable_encoded + BSON.encodeString(value);
440   } else if(value instanceof Long || Object.prototype.toString.call(value) == "[object Long]") {
441     encoded_string += BinaryParser.fromByte(BSON.BSON_DATA_LONG) + variable_encoded + BSON.encodeLong(value);
442   } else if(value.constructor == Number && value === parseInt(value, 10)) {
443     if(value > BSON.BSON_INT32_MAX || value < BSON.BSON_INT32_MIN) {      
444       encoded_string += BinaryParser.fromByte(BSON.BSON_DATA_LONG) + variable_encoded + BSON.encodeLong(Long.fromNumber(value));
445     } else {
446       encoded_string += BinaryParser.fromByte(BSON.BSON_DATA_INT) + variable_encoded + BSON.encodeInt(Integer.fromInt(value));      
447     }    
448   } else if(value.constructor == Number) {
449     encoded_string += BinaryParser.fromByte(BSON.BSON_DATA_NUMBER) + variable_encoded + BSON.encodeFloat(value);
450   } else if(Array.isArray(value)) {
451     var object_string = BSON.encodeArray(value, checkKeys);
452     encoded_string += BinaryParser.fromByte(BSON.BSON_DATA_ARRAY) + variable_encoded + BSON.encodeInt(Integer.fromInt(object_string.length + 4 + 1)) + object_string + BinaryParser.fromByte(0);
453   } else if(Object.prototype.toString.call(value) === '[object Boolean]') {
454     encoded_string += BinaryParser.fromByte(BSON.BSON_DATA_BOOLEAN) + variable_encoded + BSON.encodeBoolean(value);
455   } else if(Object.prototype.toString.call(value) === '[object Date]') {
456     encoded_string += BinaryParser.fromByte(BSON.BSON_DATA_DATE) + variable_encoded + BSON.encodeDate(value);
457   } else if(Object.prototype.toString.call(value) === '[object RegExp]') {
458     encoded_string += BinaryParser.fromByte(BSON.BSON_DATA_REGEXP) + variable_encoded + BSON.encodeRegExp(value);
459   } else if(value instanceof ObjectID || (value.id && value.toHexString) || Object.prototype.toString.call(value) == "[object ObjectID]") {
460     encoded_string += BinaryParser.fromByte(BSON.BSON_DATA_OID) + variable_encoded + BSON.encodeOid(value);
461   } else if(value instanceof Code || Object.prototype.toString.call(value) == "[object Code]") {
462     encoded_string += BinaryParser.fromByte(BSON.BSON_DATA_CODE_W_SCOPE) + variable_encoded + BSON.encodeCode(value, checkKeys);
463   } else if(value instanceof Binary || Object.prototype.toString.call(value) == "[object Binary]") {
464     var object_string = BSON.encodeBinary(value);
465     encoded_string += BinaryParser.fromByte(BSON.BSON_DATA_BINARY) + variable_encoded + BSON.encodeInt(Integer.fromInt(object_string.length - 1)) + object_string;
466   } else if(value instanceof DBRef) {
467     var object_string = BSON.encodeDBRef(value);
468     encoded_string += BinaryParser.fromByte(BSON.BSON_DATA_OBJECT) + variable_encoded + BSON.encodeInt(Integer.fromInt(object_string.length + 4 + 1)) + object_string + BinaryParser.fromByte(0);
469   } else if(Object.prototype.toString.call(value) === '[object Object]') {
470     var object_string = BSON.encodeObject(value, checkKeys);
471     encoded_string += (!top_level ? BinaryParser.fromByte(BSON.BSON_DATA_OBJECT) : '') + variable_encoded + BSON.encodeInt(Integer.fromInt(object_string.length + 4 + 1)) + object_string + BinaryParser.fromByte(0);
472   }
473 
474   return encoded_string;
475 };

501 var MACHINE_ID = (function() {

575 /**
576   Contains the a binary stream of data
577 **/
578 var Binary = exports.Binary = function(buffer) {
579   this.sub_type = BSON.BSON_BINARY_SUBTYPE_BYTE_ARRAY;
580   // Create a default sized binary in case non is passed in and prepare the size
581   if(buffer) {
582     this.buffer = buffer;    
583     this.position = buffer.length;
584   } else {
585     this.buffer = new Buffer(Binary.BUFFER_SIZE);
586     this.position = 0;
587   }
588 };
589 
590 Binary.BUFFER_SIZE = 256;
591 
592 Binary.prototype.put = function(byte_value) {
593   if(this.buffer.length > this.position) {
594     this.buffer[this.position++] = byte_value.charCodeAt(0);
595   } else {
596     // Create additional overflow buffer
597     var buffer = new Buffer(Binary.BUFFER_SIZE + this.buffer.length);
598     // Combine the two buffers together
599     this.buffer.copy(buffer, 0, 0, this.buffer.length);
600     this.buffer = buffer;
601     this.buffer[this.position++] = byte_value.charCodeAt(0);
602   }
603 };
604 
605 Binary.prototype.write = function(string, offset) {
606   offset = offset ? offset : this.position;
607   // If the buffer is to small let's extend the buffer
608   if(this.buffer.length < offset + string.length) {
609     var buffer = new Buffer(this.buffer.length + string.length);
610     this.buffer.copy(buffer, 0, 0, this.buffer.length);
611     // Assign the new buffer
612     this.buffer = buffer;
613   }
614   // Write the content to the buffer
615   this.buffer.write(string, 'binary', offset);
616   this.position = offset + string.length;
617 };
618 
619 Binary.prototype.read = function(position, length) {  
620   length = length && length > 0 ? length : this.position;  
621   // Let's extract the string we want to read
622   return this.buffer.toString('binary', position, position + length);
623 };
624 
625 Binary.prototype.value = function() {
626   return this.buffer.toString('binary', 0, this.position);
627 };
628 
629 Binary.prototype.length = function() {
630   return this.position;
