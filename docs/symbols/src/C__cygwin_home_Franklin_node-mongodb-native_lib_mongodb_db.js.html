 16 /**@constructor
 17 Exports the Db object. The Db is an object representing the database.
 18 @param databaseName the database name
 19 @param serverConfig the server configuration
 20 @param options
 21 */
 22 var Db = exports.Db = function(databaseName, serverConfig, options) {
 23   EventEmitter.call(this);
 24 
 25   this.databaseName = databaseName;
 26   this.serverConfig = serverConfig;
 27   this.options = options == null ? {} : options;
 28   // Contains all the connections for the db
 29   this.bson_serializer = this.options.native_parser ? require('../../external-libs/bson/bson') : require('./bson/bson');
 30   this.bson_deserializer = this.options.native_parser ? require('../../external-libs/bson/bson') : require('./bson/bson');  
 31   this.connections = [];
 32   // State of the db connection
 33   this.state = 'notConnected';
 34   this.pkFactory = this.options.pk == null ? this.bson_serializer.ObjectID : this.options.pk;  
 35   // Added strict
 36   this.strict = this.options.strict == null ? false : this.options.strict;
 37 };

 41 /**
 42 Opens up a connection to the server.
 43 @param callback 
 44 */
 45 Db.prototype.open = function(callback) {
 46   var self = this;
 47 
 48   // Set up connections
 49   if(self.serverConfig instanceof Server) {
 50     self.serverConfig.connection = new Connection(self.serverConfig.host, self.serverConfig.port, self.serverConfig.autoReconnect);
 51     self.connections.push(self.serverConfig.connection);
 52     var server = self.serverConfig;
 53 
 54     self.serverConfig.connection.addListener("connect", function() {
 55       // Create a callback function for a given connection
 56       var connectCallback = function(err, reply) {        
 57         if(err != null) {
 58           return callback(err, null);
 59         } else if(reply.documents[0].ismaster == 1) {
 60           self.serverConfig.master = true;
 61         } else if(reply.documents[0].ismaster == 0) {
 62           self.serverConfig.master = false;
 63         }
 64 
 65         // emit a message saying we got a master and are ready to go and change state to reflect it
 66         if(self.state == 'notConnected') {
 67           self.state = 'connected';
 68           callback(null, self);
 69         } else {
 70           callback("connection already opened");
 71         }
 72       };
 73       // Create db command and Add the callback to the list of callbacks by the request id (mapping outgoing messages to correct callbacks)
 74       var db_command = DbCommand.createIsMasterCommand(self);
 75       self.addListener(db_command.getRequestId().toString(), connectCallback);
 76       // Let's send a request to identify the state of the server
 77       this.send(db_command);
 78     });
 79 
 80     self.serverConfig.connection.addListener("data", function(message) {
 81       // Parse the data as a reply object
 82       var reply = new MongoReply(self, message);
 83       // Emit message
 84       self.emit(reply.responseTo.toString(), null, reply);
 85       // Remove the listener
 86       self.removeListener(reply.responseTo.toString(), self.listeners(reply.responseTo.toString())[0]);
 87     });
 88     
 89     self.serverConfig.connection.addListener("error", function(err) {
 90       self.emit("error", err);
 91     });
 92     
 93     // Emit timeout and close events so the client using db can figure do proper error handling (emit contains the connection that triggered the event)
 94     self.serverConfig.connection.addListener("timeout", function() { self.emit("timeout", this); });
 95     self.serverConfig.connection.addListener("close", function() { self.emit("close", this); });
 96     // Open the connection
 97     self.serverConfig.connection.open();      
 98   } else if(self.serverConfig instanceof ServerPair || self.serverConfig instanceof ServerCluster) {
 99     var serverConnections = self.serverConfig instanceof ServerPair ? [self.serverConfig.leftServer, self.serverConfig.rightServer] : self.serverConfig.servers;
100     var numberOfConnectedServers = 0;
101     serverConnections.forEach(function(server) {
102       server.connection = new Connection(server.host, server.port, server.autoReconnect);
103       self.connections.push(server.connection);
104 
105       server.connection.addListener("connect", function() {
106         // Create a callback function for a given connection
107         var connectCallback = function(err, reply) {
108           if(err != null) {
109             callback(err, null);          
110           } else {
111             if(reply.documents[0].ismaster == 1) {
112               // Locate the master connection and save it
113               self.masterConnection = server.connection;
114               server.master = true;
115             } else {
116               server.master = false;
117             }
118 
119             // emit a message saying we got a master and are ready to go and change state to reflect it
120             if(++numberOfConnectedServers == serverConnections.length && self.state == 'notConnected') {
121               self.state = 'connected';
122               callback(null, self);
123             }            
124           }
125         };
126         // Create db command and Add the callback to the list of callbacks by the request id (mapping outgoing messages to correct callbacks)
127         var db_command = DbCommand.createIsMasterCommand(self);
128         self.addListener(db_command.getRequestId().toString(), connectCallback);
129         // Let's send a request to identify the state of the server
130         this.send(db_command);
131       });
132 
133       server.connection.addListener("data", function(message) {
134         // Parse the data as a reply object
135         var reply = new MongoReply(self, message);
136         // Emit error if there is one
137         reply.responseHasError ? self.emit(reply.responseTo.toString(), reply.documents[0], reply) : self.emit(reply.responseTo.toString(), null, reply);
138         // Remove the listener
139         self.removeListener(reply.responseTo.toString(), self.listeners(reply.responseTo.toString())[0]);
140       });
141       
142       server.connection.addListener("error", function(err) {
143         self.emit("error", err);
144       });      
145 
146       // Emit timeout and close events so the client using db can figure do proper error handling (emit contains the connection that triggered the event)
147       server.connection.addListener("timeout", function() { self.emit("timeout", this); });
148       server.connection.addListener("close", function() { self.emit("close", this); });
149       // Open the connection
150       server.connection.open();
151     });
152   } else {
153     throw Error("Server parameter must be of type Server, ServerPair or ServerCluster");
154   }
155 };

183 /**
184   Get the list of all collection names for the specified db
185 **/
186 Db.prototype.collectionNames = function(collection_name, callback) {
187   if(callback == null) { callback = collection_name; collection_name = null; }
188   var self = this;
189   // Let's make our own callback to reuse the existing collections info method
190   self.collectionsInfo(collection_name, function(err, cursor) {
191     cursor.toArray(function(err, documents) {
192       // List of result documents that have been filtered
193       var filtered_documents = [];
194       // Remove any collections that are not part of the db or a system db signed with $
195       documents.forEach(function(document) {
196         if(!(document.name.indexOf(self.databaseName) == -1 || document.name.indexOf('$') != -1))
197           filtered_documents.push(document);
198       });
199       // Return filtered items
200       callback(null, filtered_documents);
201     });
202   });

244 /**
245   Evaluate javascript on the server
246 **/
247 Db.prototype.eval = function(code, parameters, callback) {
248   if(typeof parameters === "function") { callback = parameters; parameters = null; }
249   var finalCode = code;
250   var finalParameters = [];
251   // If not a code object translate to one
252   if(!(finalCode instanceof this.bson_serializer.Code)) {
253     finalCode = new this.bson_serializer.Code(finalCode);
254   }
255 
256   // Ensure the parameters are correct
257   if(parameters != null && parameters.constructor != Array) {
258     finalParameters = [parameters];
259   } else if(parameters != null && parameters.constructor == Array) {
260     finalParameters = parameters;
261   }
262   // Create execution selector
263   var selector = {'$eval':finalCode, 'args':finalParameters};
264   // Iterate through all the fields of the index
265   new Cursor(this, new Collection(this, DbCommand.SYSTEM_COMMAND_COLLECTION), selector, {}, 0, -1).nextObject(function(err, result) {
266     if(result.ok == 1) {
267       callback(null, result.retval);
268     } else {
269       callback(new Error("eval failed: " + result.errmsg), null); return;
270     }
271   });
272 };

282 /**
283   Authenticate against server
284 **/
285 Db.prototype.authenticate = function(username, password, callback) {
286   var self = this;
287   // Execute command
288   this.executeCommand(DbCommand.createGetNonceCommand(self), function(err, reply) {
289     if(err == null) {
290       // Nonce used to make authentication request with md5 hash
291       var nonce = reply.documents[0].nonce;
292       // sys.puts("=========================== NONCE: " + nonce)
293       // Execute command
294       self.executeCommand(DbCommand.createAuthenticationCommand(self, username, password, nonce), function(err, result) {
295         if(err == null && result.documents[0].ok == 1) {
296           callback(null, true);
297         } else {
298           err != null ? callback(err, false) : callback(new Error(result.documents[0].errmsg), false);
299         }
300       });      
301     } else {
302       callback(err, null);
303     }
304   });
305 };

346 /**
347   Create Collection
348 **/
349 Db.prototype.createCollection = function(collectionName, options, callback) {
350   var args = Array.prototype.slice.call(arguments, 1);
351   callback = args.pop();
352   options = args.length ? args.shift() : null;
353 
354   var self = this;
355   // Check if we have the name
356   this.collectionNames(collectionName, function(err, collections) {
357     var found = false;
358     collections.forEach(function(collection) {
359       if(collection.name == self.databaseName + "." + collectionName) found = true;
360     });
361 
362     // If the collection exists either throw an exception (if db in strict mode) or return the existing collection
363     if(found && self.strict) {
364       callback(new Error("Collection " + collectionName + " already exists. Currently in strict mode."), null); return;
365     } else if(found){
366       callback(null, new Collection(self, collectionName, self.pkFactory)); return;
367     }
368 
369     // Create a new collection and return it
370     self.executeCommand(DbCommand.createCreateCollectionCommand(self, collectionName, options), function(err, result) {
371       if(err == null && result.documents[0].ok == 1) {
372         callback(null, new Collection(self, collectionName, self.pkFactory));
373       } else {
374         err != null ? callback(err, null) : callback(new Error("Error creating collection: " + collectionName), null);
375       }
376     });
377   });
378 };

489 /**
490   Index Information
491 **/
492 Db.prototype.indexInformation = function(collectionName, callback) {
493   if(typeof collectionName === "function") { callback = collectionName; collectionName = null;}
494   // Build selector for the indexes
495   var selector = collectionName != null ? {ns: (this.databaseName + "." + collectionName)} : {};
496   var info = {};
497   // Iterate through all the fields of the index
498   new Cursor(this, new Collection(this, DbCommand.SYSTEM_INDEX_COLLECTION), selector).each(function(err, index) {
499     // Return the info when finished
500     if(index == null) {
501       callback(null, info);
502     } else {
503       info[index.name] = [];
504       for(var name in index.key) {
505         info[index.name].push([name, index.key[name]]);
506       }
507     }
508   });
509 };
